Table dim_time {
  id int [pk]
  date date
}

Table dim_state {
  id int [pk]
  state varchar
}

Table dim_county {
  id int [pk]
  county varchar
  state_id int [ref: > dim_state.id]
}

Table dim_college {
  id int [pk]
  college varchar
  ipeds_id int
  city varchar
  county_id int [ref: > dim_county.id]
}

Table dim_prison {
  id int [pk]
  facility varchar
  county_id int [ref: > dim_county.id]
}

Table fact_us_states {
  id int [pk]
  state_id int [ref: > dim_state.id]
  date_id int [ref: > dim_time.id]
  fips int
  cases int
  deaths int
}

Table fact_us_counties {
  id int [pk]
  county_id int [ref: > dim_county.id]
  date_id int [ref: > dim_time.id]
  fips int
  cases int
  deaths int
}

Table fact_us {
  id int [pk]
  date_id int [ref: > dim_time.id]
  cases int
  deaths int
}

Table fact_colleges {
  id int [pk]
  college_id int [ref: > dim_college.id]
  date_id int [ref: > dim_time.id]
  cases int
  cases_2021 int
  notes varchar
}

Table fact_prisons {
  id int [pk]
  prison_id int [ref: > dim_prison.id]
  date_id int [ref: > dim_time.id]
  staff_cases int
  prisoner_cases int
  staff_deaths int
  prisoner_deaths int
}

Table fact_mask_use {
  id int [pk]
  county_id int [ref: > dim_county.id]
  never float
  rarely float
  sometimes float
  frequently float
  always float
}

Table fact_rolling_averages {
  id int [pk]
  county_id int [ref: > dim_county.id]
  date_id int [ref: > dim_time.id]
  cases int
  deaths int
}

Table fact_excess_deaths {
  id int [pk]
  county_id int [ref: > dim_county.id]
  date_id int [ref: > dim_time.id]
  deaths int
  expected_deaths int
  excess_deaths int
}
